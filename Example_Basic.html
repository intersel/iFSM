var aStateDefinition = {
    IsDisplayed     : 
    {
		enterState:
        {
            init_function: function()
            {
            	this.myUIObject.html("I'm Displayed");
            }
        },
        click:   
        {
            next_state: 'IsNotDisplayed'
        }
    }, 
    IsNotDisplayed      : 
    {
        enterState:   
        {
            init_function: function()
            {
            	this.myUIObject.html("I'm NOT Displayed");
            }
        },
        click:   
        {
            next_state: 'IsDisplayed'
        }
    },
    DefaultState        :
    {
        start   :
        {
            next_state  : 'IsDisplayed'
        }
    }
};

$(document).ready(function() {
    $('#myButton').iFSM(aStateDefinition);
});
